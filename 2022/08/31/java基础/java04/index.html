public int[] addNum(int num, int[] arr) {
// TODO Auto-generated method stub

    //获取新数组
    int [] newArr = new int [arr.length + 1];
    for (int i = 0; i < arr.length; i++) {
        newArr[i] = arr[i];//把旧数组放到新数组
    }
    newArr[newArr.length - 1] = num;//newArr.length - 1是数组的最后一位数
    return newArr;
}
